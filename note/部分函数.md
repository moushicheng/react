export function lanesToEventPriority(lanes: Lanes): EventPriority {
  //获取lanes中从右往左数的第一个出现1的优先级
  const lane = getHighestPriorityLane(lanes); // getHighestPriorityLane：lanes & -lanes;

  //isHigherEventPriority: return a !== 0 && a < b; 看哪个二进制位的数目更大咯
  if (!isHigherEventPriority(DiscreteEventPriority, lane)) {
    //逻辑是:如果DiscreteEventPriority>=lane
    return DiscreteEventPriority; //10
  }
  if (!isHigherEventPriority(ContinuousEventPriority, lane)) {
    //逻辑是:如果ContinuousEventPriority>=lane
    return ContinuousEventPriority; ////1000
  }
  if (includesNonIdleWork(lane)) { 
    //(lanes & NonIdleLanes) !== NoLanes; 和逻辑
    return DefaultEventPriority;
  }
  return IdleEventPriority;
}
